016package edu.cnm.deepdive;
017
018import static org.junit.jupiter.api.Assertions.*;
019
020import org.junit.jupiter.api.DisplayName;
021import org.junit.jupiter.params.ParameterizedTest;
022import org.junit.jupiter.params.provider.CsvFileSource;
023
024class DNATest {
025
026  @DisplayName("DNA.complement(String)")
027  @ParameterizedTest(name = "[{index}] sequence = {0}, expected = {1}")
028  @CsvFileSource(resources = "sequences-upper.csv")
029  void complement(String sequence, String expected) {
030    assertEquals(expected, DNA.complement(sequence));
031  }
032
033  @DisplayName("DNA.complement(String) & DNA.complement(String, false)")
034  @ParameterizedTest(name = "[{index}] sequence = {0}, expected = {1}")
035  @CsvFileSource(resources = "sequences-mixed-disrespect.csv")
036  void complementDisrespectCase(String sequence, String expected) {
037    assertAll(
038        () -> assertEquals(expected, DNA.complement(sequence)),
039        () -> assertEquals(expected, DNA.complement(sequence, false))
040    );
041  }
042
043  @DisplayName("DNA.Complement(String, true)")
044  @ParameterizedTest(name = "[{index}] sequence = {0}, expected = {1}")
045  @CsvFileSource(resources = "sequences-mixed-respect.csv")
046  void complementRespectCase(String sequence, String expected) {
047    assertEquals(expected, DNA.complement(sequence, true));
048  }
049
050}
