001package edu.cnm.deepdive;
002
003import static org.junit.jupiter.api.Assertions.*;
004
005import org.junit.jupiter.params.ParameterizedTest;
006import org.junit.jupiter.params.provider.CsvFileSource;
007
008class DNATest {
009
010  @ParameterizedTest
011  @CsvFileSource(resources = "sequences_upper.csv")
012  void complement(String sequence, String expected) {
013    assertEquals(expected, DNA.complement(sequence));
014  }
015
016  @ParameterizedTest
017  @CsvFileSource(resources = "sequences_mixed_disrespect.csv")
018  void complementDisrespectCase(String sequence, String expected) {
019    assertAll(
020        () -> assertEquals(expected, DNA.complement(sequence)),
021        () -> assertEquals(expected, DNA.complement(sequence, false))
022    );
023  }
024
025  @ParameterizedTest
026  @CsvFileSource(resources = "sequences_mixed_respect.csv")
027  void complementRespectCase(String sequence, String expected) {
028    assertEquals(expected, DNA.complement(sequence, true));
029  }
030
031}
