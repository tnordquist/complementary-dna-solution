001package edu.cnm.deepdive;
002
003import static org.junit.jupiter.api.Assertions.*;
004
005import org.junit.jupiter.api.DisplayName;
006import org.junit.jupiter.params.ParameterizedTest;
007import org.junit.jupiter.params.provider.CsvFileSource;
008
009class DNATest {
010
011  @DisplayName("DNA.complement(String)")
012  @ParameterizedTest(name = "[{index}] sequence = {0}, expected = {1}")
013  @CsvFileSource(resources = "sequences-upper.csv")
014  void complement(String sequence, String expected) {
015    assertEquals(expected, DNA.complement(sequence));
016  }
017
018  @DisplayName("DNA.complement(String) & DNA.complement(String, false)")
019  @ParameterizedTest(name = "[{index}] sequence = {0}, expected = {1}")
020  @CsvFileSource(resources = "sequences-mixed-disrespect.csv")
021  void complementDisrespectCase(String sequence, String expected) {
022    assertAll(
023        () -> assertEquals(expected, DNA.complement(sequence)),
024        () -> assertEquals(expected, DNA.complement(sequence, false))
025    );
026  }
027
028  @DisplayName("DNA.Complement(String, true)")
029  @ParameterizedTest(name = "[{index}] sequence = {0}, expected = {1}")
030  @CsvFileSource(resources = "sequences-mixed-respect.csv")
031  void complementRespectCase(String sequence, String expected) {
032    assertEquals(expected, DNA.complement(sequence, true));
033  }
034
035}
