001package edu.cnm.deepdive;
002
003/**
004 * This class implements a simple string converter that replaces the nucleobase symbols
005 * (<code>A</code>,<code>T</code>, <code>C</code>, and <code>G</code>), with their respective
006 * complements (<code>T</code>,<code>A</code>, <code>G</code>, and <code>C</code>), in a string
007 * representing a DNA sequence/subsequence.
008 */
009public class DNA {
010
011  private DNA() {
012    // NOTE: There is NO need to do anything with this constructor! The methods you will implement
013    // in this class is static; thus, there is no need to create instances of this class.
014  }
015
016  /* BASIC PROBLEM */
017
018  /**
019   * Converts and returns the DNA complement of <code>sequence</code>, with each nucleobase symbol
020   * (character) replaced by its complementary symbol.
021   *
022   * @param sequence DNA sequence.
023   * @return complementary DNA sequence.
024   */
025  public static String complement(String sequence) {
026    return complement(sequence, false);
027  }
028
029  /* EXTRA CREDIT PROBLEM */
030
031  /**
032   * Converts and returns the DNA complement of <code>sequence</code>, with each nucleobase symbol
033   * (character) replaced by its complementary symbol. Optionally, the case of <code>sequence</code>
034   * is respected, according <code>respectCase</code>: if <code>true</code>, each character in
035   * <code>sequence</code> is replaced by its complement of the same case; if <code>false</code>,
036   * all characters in the returned value will be upper-case, regardless of the case of the
037   * characters in <code>sequence</code>.
038   *
039   * @param sequence DNA sequence.
040   * @param respectCase flag controlling the case of the returned value.
041   * @return complementary DNA sequence.
042   */
043  public static String complement(String sequence, boolean respectCase) {
044    StringBuilder builder = new StringBuilder();
045    char[] chars = sequence.toCharArray();
046    for (char c : chars) {
047      boolean isLowerCase = respectCase && Character.isLowerCase(c);
048      c = Character.toUpperCase(c);
049      char complement;
050      switch (c) {
051        case 'A':
052          complement = 'T';
053          break;
054        case 'C':
055          complement = 'G';
056          break;
057        case 'G':
058          complement = 'C';
059          break;
060        case 'T':
061          complement = 'A';
062          break;
063        default:
064          complement = c;
065          break;
066      }
067      if (isLowerCase) {
068        complement = Character.toLowerCase(complement);
069      }
070      builder.append(complement);
071    }
072    return builder.toString();
073  }
074
075}
