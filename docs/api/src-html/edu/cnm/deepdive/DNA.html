016package edu.cnm.deepdive;
017
018/**
019 * This class includes the {@link #complement(String)} and {@link #complement(String, boolean)}
020 * methods, both of which implement a simple string converter that replaces the nucleobase symbols
021 * (<code>A</code>,<code>T</code>, <code>C</code>, and <code>G</code>), with their respective
022 * complements (<code>T</code>,<code>A</code>, <code>G</code>, and <code>C</code>), in a string
023 * representing a DNA sequence/subsequence. Implementation of these methods is included in the
024 * practical exam materials of the Deep Dive Coding Java + Android Bootcamp.
025 */
026public class DNA {
027
028  private DNA() {
029    // NOTE: There is NO need to do anything with this constructor! The methods you will implement
030    // in this class is static; thus, there is no need to create instances of this class.
031  }
032
033  /* BASIC PROBLEM */
034
035  /**
036   * Converts and returns the DNA complement of <code>sequence</code>, with each nucleobase symbol
037   * (character) replaced by its complementary symbol.
038   *
039   * @param sequence DNA sequence.
040   * @return complementary DNA sequence.
041   */
042  public static String complement(String sequence) {
043    return complement(sequence, false);
044  }
045
046  /* EXTRA CREDIT PROBLEM */
047
048  /**
049   * Converts and returns the DNA complement of <code>sequence</code>, with each nucleobase symbol
050   * (character) replaced by its complementary symbol. Optionally, the case of <code>sequence</code>
051   * is respected, according <code>respectCase</code>: if <code>true</code>, each character in
052   * <code>sequence</code> is replaced by its complement of the same case; if <code>false</code>,
053   * all characters in the returned value will be upper-case, regardless of the case of the
054   * characters in <code>sequence</code>.
055   *
056   * @param sequence DNA sequence.
057   * @param respectCase flag controlling the case of the returned value.
058   * @return complementary DNA sequence.
059   */
060  public static String complement(String sequence, boolean respectCase) {
061    StringBuilder builder = new StringBuilder();
062    char[] chars = sequence.toCharArray();
063    for (char c : chars) {
064      boolean isLowerCase = respectCase && Character.isLowerCase(c);
065      c = Character.toUpperCase(c);
066      char complement;
067      switch (c) {
068        case 'A':
069          complement = 'T';
070          break;
071        case 'C':
072          complement = 'G';
073          break;
074        case 'G':
075          complement = 'C';
076          break;
077        case 'T':
078          complement = 'A';
079          break;
080        default:
081          complement = c;
082          break;
083      }
084      if (isLowerCase) {
085        complement = Character.toLowerCase(complement);
086      }
087      builder.append(complement);
088    }
089    return builder.toString();
090  }
091
092}
